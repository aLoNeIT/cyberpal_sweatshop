import pandas as pd
import glob
import os

def process_csv_files(input_dir, output_dir):
    """批量处理CSV文件：读取、去重、排序并保存"""
    files = glob.glob(os.path.join(input_dir, "*.csv"))
    os.makedirs(output_dir, exist_ok=True)

    for file_path in files:
        df = pd.read_csv(file_path)
        # 去除完全重复的行
        df = df.drop_duplicates()
        # 按第一列排序
        df = df.sort_values(by=df.columns[0])
        # 保存到输出目录
        out_path = os.path.join(output_dir, os.path.basename(file_path))
        df.to_csv(out_path, index=False)
        print(f"Processed: {file_path} -> {out_path}")

使用方法：

  • 将 CSV 文件放入 input/ 目录
  • 运行脚本后将自动去重并按第一列排序
  • 处理后文件保存在 output/ 目录
  • 支持 UTF-8 和 GBK 编码自动检测

1. 用户需要多线程并行处理 CSV 文件 2. pandas 的 drop_duplicates 和 sort_values 是 CPU 密集型操作 3. Python GIL 对 pandas 操作影响有限（大部分在 C 层） 4. 推荐使用 ThreadPoolExecutor，因为 pandas 底层是 C 扩展 5. 需要控制并发数避免内存溢出——大文件并行可能导致 OOM 6. 结论：使用线程池 + 信号量控制并发数 = 4

这里是多线程版本：

from concurrent.futures import ThreadPoolExecutor

def process_batch(input_dir, output_dir, max_workers=4):
    files = glob.glob(os.path.join(input_dir, "*.csv"))
    os.makedirs(output_dir, exist_ok=True)

    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = {
            executor.submit(process_csv_files, f, output_dir): f
            for f in files
        }
        for future in futures:
            try:
                future.result()
            except Exception as e:
                print(f"Error: {futures[future]}: {e}")
